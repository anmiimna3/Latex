\begin{enumerate}[label= \textbf{(\textit{\alph*})}]
    \item
        قرار دهید $s = (\lambda xy . yx)$:
        \begin{gather*}
            \begin{split}
                stu & = (\lambda xy. yx) tu \\
                    & = ut 
            \end{split}
        \end{gather*}
    \item 
        قرار دهید $s = (\lambda x. y)$ به طوری که $x \ne y$. دقت کنید که داریم: $(\lambda x.y) s = y$:
        \begin{gather*}
                st  = (\lambda x. y) t  = y = (\lambda x. y) s = ss 
        \end{gather*}
\end{enumerate}
