\begin{enumerate}[label=\ilabel]
    \item
        Let $\braket{s_1, s_2, \dots, s_n}$ be the generators of $M_1$. 
        \begin{gather*}
            0 \to M_1 \overset{\varphi}{\relbar \joinrel \to} M \overset{\psi}{\relbar \joinrel \to} M_2 \to 0
        \end{gather*}
        We know that $ker \psi = im \varphi$. Therefore we can generate $ker \psi$ with $\varphi(s_1), \varphi(s_2), \dots, \varphi(s_n)$. Now since $\psi$ is surjective, we have that:
        \begin{gather*}
            \frac{M}{ker \psi} = M_2
        \end{gather*}
        Since $M_2$ is finitely generated, then $M/ker\psi$ is also finitely generated. Therefore the set of cosets is finitely generated. Let the generators be $\braket{\overline{t_1}, \overline{t_2}, \dots, \overline{t_k}}$. We argue that $\braket{\varphi(s_1), \dots, \varphi(s_n), t_1, \dots, t_k}$ generates $M$. Let $m \in M$. $m$ belongs to one of cosets, thus we can write:
        \begin{gather*}
            m + ker \psi = r_1 \overline{t_1} + \dots + r_k \overline{t_k} + ker\psi\\
            \implies m - r_1 {t_1} - \dots - r_k {t_k} = i
        \end{gather*}
        for some $i$ in $ker \psi$. Since $ker \psi = im \varphi$, then we can write:
        \begin{gather*}
            i = u_1 \varphi(s_1) + \dots + u_n \varphi(s_n)
        \end{gather*}
        This gives us that $m$ is generated by the generator we introduced:
        \begin{gather*}
            m = r_1 t_1 + \dots + r_k t_k + u_1 \varphi(s_1) + \dots + u_n\varphi(s_n)
        \end{gather*}
        Thus $M$ is finitely generated.

        However it is not necessarily true that if $M$ is finitely generated, then $M_1$ and $M_2$ are finitely generated as well. An example of this could be the $R$-modules below:
        \begin{gather*}
            0 \to R[x_1, x_2, \dots] \overset{\sigma_0}{\relbar \joinrel \to} R \overset{id}{\relbar \joinrel \to} R \to 0
        \end{gather*}
        Where $R$ has 1, and $\sigma_0$ is valuation at 0 for all $x_i$s.
    \item 
        

\end{enumerate}
